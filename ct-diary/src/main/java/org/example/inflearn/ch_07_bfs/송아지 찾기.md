import java.util.*;

class Solution {
    public int solution(int[] arr) {
        int[] di = {1, 5, -1};
        ArrayDeque<Integer> q = new ArrayDeque<>();
        HashSet<Integer> visited = new HashSet<>();

        int start = arr[0];
        int end = arr[1];

        q.addLast(start);
        visited.add(start);

        int level = 0;

        while(!q.isEmpty()) {
            int size = q.size();
            
            // 반드시 q의 자식 노드의 갯수만큼 반복 필요             
            for(int i = 0; i < size; i++) {
                int now = q.pollFirst();

                if(now == end) return level;

                for(int d : di) {
                    
                    // 다음 노드의 값들, 만약에 여기서 return 하고 싶으면 +1이 필요함         
                    int next = now + d;
                    if(!visited.contains(next)) {
                        q.addLast(next);
                        visited.add(next);
                    }
                }
            }
            level++;
        }

        return -1; // 도달 불가 시
    }

    public static void main(String[] args) {
        Solution T = new Solution();
        System.out.println(T.solution(new int[]{5, 14})); // 3
        System.out.println(T.solution(new int[]{0, 11})); // 3
        System.out.println(T.solution(new int[]{5, 5}));  // 0
    }
}
